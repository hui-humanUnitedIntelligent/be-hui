HUI 042 — Schema Reality Check

Führe jeden Block im Supabase SQL Editor aus. Jeder Block ist unabhängig.

1. experiences — existiert die Tabelle?

SELECT EXISTS (
  SELECT 1 FROM information_schema.tables
  WHERE table_schema = 'public' AND table_name = 'experiences'
) AS table_exists;

2. experiences — welche Columns existieren WIRKLICH?

SELECT column_name, data_type, column_default, is_nullable
FROM information_schema.columns
WHERE table_schema = 'public' AND table_name = 'experiences'
ORDER BY ordinal_position;

3. Spezifisch: booking_mode vorhanden?

SELECT
  CASE WHEN EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_schema = 'public'
    AND table_name = 'experiences'
    AND column_name = 'booking_mode'
  ) THEN 'booking_mode: ✓ VORHANDEN'
  ELSE 'booking_mode: ✗ FEHLT — Migration 038 nicht ausgeführt!'
  END AS status,
  CASE WHEN EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_schema = 'public'
    AND table_name = 'experiences'
    AND column_name = 'pricing_type'
  ) THEN 'pricing_type: ✓ VORHANDEN'
  ELSE 'pricing_type: ✗ FEHLT — Migration 039 nicht ausgeführt!'
  END AS status2,
  CASE WHEN EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_schema = 'public'
    AND table_name = 'experiences'
    AND column_name = 'mood_tags'
  ) THEN 'mood_tags: ✓ VORHANDEN'
  ELSE 'mood_tags: ✗ FEHLT — Migration 039 nicht ausgeführt!'
  END AS status3;

4. RLS Policies auf experiences

SELECT policyname, cmd, qual
FROM pg_policies
WHERE tablename = 'experiences'
ORDER BY policyname;

5. Alle experiences in der DB

SELECT id, title, status, visibility, user_id, created_at
FROM public.experiences
ORDER BY created_at DESC
LIMIT 20;
